Yielding other iterables

Yield what you like!
  • Generators can yield other iterables
  • ...even other generators
const letters = [ 'a', 'b', 'c' ];

      
function *numbers() {{ '{' }}
  yield 1;
  yield 2;
  yield 3;
}

      
function *alphaNumericGenerator() {{ '{' }}
  yield* letters;
  yield* numbers();
}

      
const numbersAndLetters = alphaNumericGenerator();

      
[ ...numbersAndLetters ]
// [ 'a', 'b', 'c', 1, 2, 3 ]
