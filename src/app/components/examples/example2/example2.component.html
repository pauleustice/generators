Iterating

It's an Iterable!
  • You can spread a generator object like an Array, String or other Iterable
function* numbersGenerator() {{ '{' }}
  yield 3;
  yield 2;
  yield 1;
}

      
const countdown = numbersGenerator();

      
[ 5, 4, ...countdown ]
// [ 5, 4, 3, 2, 1 ]
  • Or use it in for...of loops
function* lettersGenerator() {{ '{' }}
  yield 'a';
  yield 'b';
  yield 'c';
}

      
const alphabet = lettersGenerator();

      
for (letter of alphabet) {{ '{' }}
  console.log(letter);
}
// a, b, c
