Yielding values

A basic example
  • Calling next() yields the next value within the generator
function* threeTwoOne() {{ '{' }}
  yield 3;
  yield 2;
  yield 1;
}

      
const countdown = threeTwoOne();

      
countdown.next();
// {{ '{' }} value: 3, done: false }
countdown.next();
// {{ '{' }} value: 2, done: false }
countdown.next();
// {{ '{' }} value: 1, done: false }
countdown.next();
// {{ '{' }} value: undefined, done: true }
